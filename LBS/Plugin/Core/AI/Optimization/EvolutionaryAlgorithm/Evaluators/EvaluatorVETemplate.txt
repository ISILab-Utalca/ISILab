using ISILab.AI.Categorization;
using ISILab.LBS.CustomComponents;
using ISILab.LBS.Editor;
using UnityEditor;
using UnityEngine.UIElements;

namespace ISILab.LBS.VisualElements
{
    [LBSCustomEditor("#TARGETNAME#", typeof(#TARGETNAME#))]
    public class #TARGETNAME#VE : LBSCustomEditor
    {
		LBSCustomButton configurationButton;

		public #TARGETNAME#VE(object target) : base(target)
		{
			CreateVisualElement();
			SetInfo(target);
		}

		protected override VisualElement CreateVisualElement()
		{
			configurationButton = new LBSCustomButton();
			configurationButton.style.height = 30;
			configurationButton.text = "Advanced configuration";
			configurationButton.clicked -= ShowConfiguration;
			configurationButton.clicked += ShowConfiguration;
       		Add(configurationButton);

			static void ShowConfiguration() => Selection.activeObject = #TARGETNAME#.config;

			return this;
		}

		public override void SetInfo(object paramTarget) { }
    }
}
